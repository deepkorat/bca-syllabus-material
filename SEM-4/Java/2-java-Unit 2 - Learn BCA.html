import java.util.*;
class ch2_p1
{
public static void main(String[] args)
{
    int tmp,i,j;
    int n[]=new int[3];
    System.out.println("Enter values in array");
    Scanner sc=new Scanner(System.in);
    for(i=0;i<3;i++)
    {
        n[i]=sc.nextInt();
    }
    System.out.println("Unsorted Array list ");
    for(i=0;i<3;i++)
    {
        System.out.println(n[i] + "\t");
    }
    for(i=1;i<3;i++)
    {
        for(j=0;j<3;j++)
        {
            if (n[i] < n[j])
            {
               tmp = n[i];
               n[i] = n[j];
               n[j] = tmp;
            }
    }
    }
    System.out.println("sorted Array list ");
    for(i=0;i<3;i++)
    {
        System.out.print(n[i] + "\t");
    }
}

interface numbers
{
    int process(int x,int y);
}
class Sum implements numbers
{
    int z;
    public int process(int x,int y)
    {
        //this.x=x;
        z=x+y;
        System.out.println("the value of z is::->"+z);
        return z;
    }
}
class Average implements numbers
{
    int tot,avg;
    public int process(int x,int y)
    {
        tot=x+y;
        avg=tot/2;
        System.out.println("the average is::->"+avg);
        return avg;
    }
}

interface classify

class result implements exam,classify
{
    public boolean Pass(int mark)

        {
            System.out.println("Pass");
        }

        {
            System.out.println("Fail");
        }
    return true;

    {
    if(average>=60)
    {
        System.out.println("First");
    }

    {
        System.out.println("Second");
    }
    else
    {
        System.out.println("No division");
    }
    return "pass";
    }

abstract class calculation
{
    double height,width;
    calculation(double height,double width)
    {
        this.height=height;
        this.width=width;
    }
    abstract double area();
}
class rectangle extends calculation
{
    rectangle(double h,double w)
    {
        super(h,w);
    }
    double area()
    {
        return height*width;
    }
}
class triangle extends calculation
{
    triangle(double h,double w)
    {
        super(h,w);
    }
    double area()
    {
        return 0.5*height*width;
    }
}
class two
{
    public static void main(String ar[])
    {
        calculation cl;
        rectangle r=new rectangle(10,15);
        cl=r;
        System.out.println("area of rectangle is "+cl.area());
        triangle t=new triangle(15,20);
        cl=t;
        System.out.println("area of triangle is "+cl.area());
    }
}
