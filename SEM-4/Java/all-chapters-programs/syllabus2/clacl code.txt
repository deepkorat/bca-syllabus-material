            Double result;

            
            
            if(operation == "+")
            {
               result = firstnum + secondnum;
               textBox1.Text = Convert.ToString(result);
               firstnum = result;
            }
            if (operation == "-")
            {
                result = firstnum - secondnum;
                textBox1.Text = Convert.ToString(result);
                firstnum = result;
            }
            if (operation == "*")
            {
                result = firstnum * secondnum;
                textBox1.Text = Convert.ToString(result);
                firstnum = result;
            }
            if (operation == "/")
            {
                if (secondnum == 0)
                {
                    textBox1.Text = "Can not divied by 0";
                }
                else
                {
                    result = firstnum / secondnum;
                    textBox1.Text = Convert.ToString(result);
                    firstnum = result;
                }
            }
            if (operation == "X^N")
            {
                textBox1.Text = Convert.ToString(Math.Pow(firstnum, secondnum));
            }
            
            

        }
